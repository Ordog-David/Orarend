SQL Feladat

2vas: SELECT letszam FROM megye WHERE nev = "Vas";

3somogy: SELECT SUM(aerob.letszam) FROM aerob JOIN megye ON megye.kod = aerob.mkod WHERE megye.nev = "Somogy";

4zala: SELECT aerob.letszam FROM aerob JOIN megye ON megye.kod = aerob.mkod JOIN allapot ON allapot.kod = aerob.allkod WHERE megye.nev = "Zala" AND aerob.nem = 1 AND allapot.nev = "Egészséges";

5heves: SELECT COUNT(*) FROM megye WHERE letszam < (SELECT letszam FROM megye WHERE nev = "Heves");

6pest: SELECT SUM(aerob.letszam)/megye.letszam FROM aerob JOIN megye ON megye.kod = aerob.mkod WHERE megye.nev = "Pest" GROUP BY mkod;

7egesz: SELECT megye.nev, aerob.letszam FROM aerob JOIN megye ON megye.kod = aerob.mkod JOIN allapot ON allapot.kod = aerob.allkod WHERE aerob.nem = 0 AND allapot.nev = "Egészséges" ORDER BY aerob.letszam DESC;

8arany: SELECT megye.nev, SUM(aerob.letszam)/megye.letszam AS arany FROM aerob JOIN megye ON megye.kod = aerob.mkod GROUP BY mkod ORDER BY arany DESC LIMIT 1;

9negyed: SELECT megye.nev AS Megyenév, SUM(aerob.letszam)/megye.letszam AS Arány FROM aerob JOIN megye ON megye.kod = aerob.mkod JOIN allapot ON allapot.kod = aerob.allkod WHERE allapot.nev = "fejlesztést igényel" OR allapot.nev = "fokozott fejlesztést igényel" GROUP BY mkod;
